180 return 1;

184 case echoJitterNone:
185 case echoJitterRandom:
186 break;
187 case echoJitterGrid:
188 case echoJitterJitter:
189 tmp = (int)sqrt(parm->numSamples);
190 if (tmp*tmp != parm->numSamples) {

210echoChannelAverage(echoCol_t *img,

212 int s;

214

217 R += tstate->chanBuff[0 + ECHO_IMG_CHANNELS5*s];
218 G += tstate->chanBuff[1 + ECHO_IMG_CHANNELS5*s];
219 B += tstate->chanBuff[2 + ECHO_IMG_CHANNELS5*s];
220 A += tstate->chanBuff[3 + ECHO_IMG_CHANNELS5*s];
221 T += tstate->chanBuff[4 + ECHO_IMG_CHANNELS5*s];
222 }
223 img[0] = R / parm->numSamples;
224 img[1] = G / parm->numSamples;
225 img[2] = B / parm->numSamples;
226 img[3] = A / parm->numSamples;
227 img[4] = T;
228
229 return;

422** top-level call to accomplish all (ray-tracing) rendering. As much

427echoRTRender(Nrrd *nraw, limnCamera *cam, echoScene *scene,
428 echoRTParm *parm, echoGlobalState *gstate) {
429 static const char me[]="echoRTRender";
430 int tid, ret;
431 airArray *mop;
432 echoThreadState *tstate[ECHO_THREAD_MAX512];
433
434 if (echoRTRenderCheck(nraw, cam, scene, parm, gstate)) {
435 biffAddf(ECHOechoBiffKey, "%s: problem with input", me);
436 return 1;
437 }
438 gstate->nraw = nraw;
439 gstate->cam = cam;
440 gstate->scene = scene;
441 gstate->parm = parm;
442 mop = airMopNew();
443 if (nrrdMaybeAlloc_va(nraw, echoCol_ntnrrdTypeFloat, 3,
444 AIR_CAST(size_t, ECHO_IMG_CHANNELS)((size_t)(5)),
445 AIR_CAST(size_t, parm->imgResU)((size_t)(parm->imgResU)),
446 AIR_CAST(size_t, parm->imgResV)((size_t)(parm->imgResV)))) {
447 biffMovef(ECHOechoBiffKey, NRRDnrrdBiffKey, "%s: couldn't allocate output image", me);
448 airMopError(mop); return 1;
449 }
450 airMopAdd(mop, nraw, (airMopper)nrrdNix, airMopOnError);
451 nrrdAxisInfoSet_va(nraw, nrrdAxisInfoLabel,
452 "r,g,b,a,t", "x", "y");
453 nrrdAxisInfoSet_va(nraw, nrrdAxisInfoMin,
454 AIR_NAN(airFloatQNaN.f), cam->uRange[0], cam->vRange[0]);
455 nrrdAxisInfoSet_va(nraw, nrrdAxisInfoMax,
456 AIR_NAN(airFloatQNaN.f), cam->uRange[1], cam->vRange[1]);
457 gstate->time = airTime();
458
459 if (parm->numThreads > 1) {
460 gstate->workMutex = airThreadMutexNew();
461 airMopAdd(mop, gstate->workMutex,
462 (airMopper)airThreadMutexNix, airMopAlways);
463 } else {
464 gstate->workMutex = NULL((void*)0);
465 }
466 for (tid=0; tid<parm->numThreads; tid++) {
467 if (!( tstate[tid] = echoThreadStateNew() )) {
468 biffAddf(ECHOechoBiffKey, "%s: failed to create thread state %d", me, tid);
469 airMopError(mop); return 1;

471 if (echoThreadStateInit(tid, tstate[tid], parm, gstate)) {
472 biffAddf(ECHOechoBiffKey, "%s: failed to initialized thread state %d", me, tid);
473 airMopError(mop); return 1;
